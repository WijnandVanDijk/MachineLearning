class Neuron:
    
    def __init__(self, bias, weight):
        self.weight = weight
        self.bias = bias
        
        self.output = 0
        self.error = 0
        self.output = 0
        self.learning_rate = 0.1
        
    def activate(self, x):
        return 1/(1+2.71828**(-x))
    
    def get_output(self, _input):
        som = 0
        for i in range(len(_input)):
            som += _input[i] * self.weight[i]
        x = som + self.bias
        self.output = self.activate(x)
        return self.output
    
    def error(self, target):
        e = self.output * (1-self.output) * -(target - self.output)
        self.error = e
        return e
    
    def gradient(self):
        return self.output * self.error
    
    def update(self, _input, target, runs):

        for y in range(runs):
            y = self.get_output(_input)
            print(y)

            error = target - y

            for i in range(len(self.weight)):
                delta = self.learning_rate * error * _input[i]
                self.weight[i] = delta + self.weight[i]
            
            delta_bias = self.learning_rate * error
            self.bias = self.bias + delta_bias
            
    def error_hidden(self, prev_neurons, neuron_count):
        weighted_errors = []
        for neuron in prev_neurons.neurons:
            weighted_error = neuron.error * neuron.weight[neuron_count]
            weighted_errors.append(weighted_error)
        som_errors = sum(weighted_errors)

        sigmoid = self.output * (1 - self.output)

        self.error = sigmoid * som_errors

        return self.error
        
    def __str__(self):
        return ('Bias: {}' + ' Weight: {}').format(self.bias, self.weight)
